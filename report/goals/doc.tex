\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
    <target name="doc" depends="build">
        <echo message="Generating JavaDoc documentation..."/>

        <mkdir dir="${fse.doc.dir}"/>
        <javadoc sourcepath="${fse.compile.source}"
                 destdir="${fse.doc.dir}"
                 source="${fse.compile.jvm.source}"
                 windowtitle="${fse.doc.windowtitle}"
                 header="${fse.doc.header}"
                 encoding="${fse.doc.encoding}"
                 charset="${fse.doc.encoding}"
                 docencoding="${fse.doc.encoding}"
                 failonerror="false"
                 access="private">
            <packageset dir="${fse.compile.source}" defaultexcludes="yes">
                <include name="weblab/**"/>
            </packageset>
            <classpath>
                <path refid="fse.ivy.path.compile"/>
            </classpath>
        </javadoc>

        <copy todir="${fse.build.target}/docs/javadoc">
            <fileset dir="${fse.doc.dir}"/>
        </copy>

        <checksum algorithm="MD5" totalproperty="project.md5.hash">
            <fileset dir="${fse.compile.source}" includes="**/*.java" excludes="test/**"/>
        </checksum>

        <checksum algorithm="SHA-1" totalproperty="project.sha1.hash">
            <fileset dir="${fse.compile.source}" includes="**/*.java" excludes="test/**"/>
        </checksum>

        <copy todir="${fse.build.target}/docs/javadoc">
            <fileset dir="${fse.doc.dir}"/>
        </copy>

        <tstamp/>

        <jar destfile="${fse.build.target}/${fse.build.name}-${fse.build.version}.war"
            update="true"
            basedir="${fse.build.target}"
            includes="docs/**">
            <manifest>
                <attribute name="Documentation" value="JavaDoc included in docs/javadoc"/>
                <attribute name="Doc-Generated" value="${TODAY}"/>
                <attribute name="Doc-Encoding" value="${fse.doc.encoding}"/>
                <attribute name="Project-MD5-Checksum" value="${project.md5.hash}"/>
                <attribute name="Project-SHA1-Checksum" value="${project.sha1.hash}"/>
            </manifest>
        </jar>

        <echo message="Done task 'doc'!" />
    </target>
\end{lstlisting}

\subsubsection{Комментарии}

Генерирует JavaDoc-документацию по всем классам проекта, вычисляет контрольные суммы MD5 и SHA-1 исходных файлов, а затем добавляет документацию и хеши в WAR-архив. JavaDoc создаётся для всех пакетов \texttt{weblab.*} (документируются все классы, поля и методы). Вычисленные хеши сохраняются как суммарные значения для всех Java-файлов и добавляются в \texttt{MANIFEST.MF} WAR-архива в виде атрибутов. Генерация JavaDoc не останавливает сборку при ошибках, что позволяет продолжить выполнение даже при наличии проблем в комментариях. Цель зависит от build, так как требует существующего WAR-архива для обновления.
