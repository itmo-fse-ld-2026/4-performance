<target name="scp" depends="build">
  <echo message="Copying project to the ${fse.scp.ssh.host}:${fse.scp.file.todir}..." />

  <scp todir="${fse.scp.ssh.host}:${fse.scp.file.todir}"
       port="${fse.scp.ssh.port}"
       keyfile="/ssh/${fse.scp.ssh.key}"
       trust="true">
    <fileset dir="target">
      <filename regex="${fse.scp.file.regex}"/>
    </fileset>
  </scp>
</target>
\end{lstlisting}
