\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
<target name="alt">
  <delete dir="${fse.alt.dir}" failonerror="false"/>
  <mkdir dir="${fse.alt.dir}"/>

  <echo message="Creating alternative version of project..." />

  <copy todir="${fse.alt.dir}">
    <fileset dir="src"/>
  </copy>

  <for list="${fse.alt.pairs}" param="pair" delimiter=",">
    <sequential>
      <local name="from"/>
      <local name="to"/>

      <propertyregex property="from"
                      input="@{pair}"
                      regexp="^(.+):"
                      select="\1" />
      <propertyregex property="to"
                      input="@{pair}"
                      regexp=":(.+)$"
                      select="\1" />
      <replaceregexp match="${from}" replace="${to}" flags="g" byline="true">
        <fileset dir="${fse.alt.dir}">
          <filename regex="${fse.alt.files}"/>
        </fileset>
      </replaceregexp>
    </sequential>
  </for>

  <for list="${fse.alt.pairs}" param="pair" delimiter=",">
    <sequential>
      <local name="from"/>
      <local name="to"/>

      <propertyregex property="from"
                      input="@{pair}"
                      regexp="^(.+):"
                      select="\1" />
      <propertyregex property="to"
                      input="@{pair}"
                      regexp=":(.+)$"
                      select="\1" />
      
      <move todir="${fse.alt.dir}">
        <fileset dir="${fse.alt.dir}">
          <include name="**/*"/>
        </fileset>
        <mapper type="regexp" from="^(.*)${from}(.*)$" to="\1${to}\2"/>
      </move>
    </sequential>
  </for>

  <antcall target="build">
    <param name="fse.compile.source" value="${fse.alt.dir}" />

    <param name="fse.build.version" value="${fse.build.version}-alt" />
  </antcall>
</target>
\end{lstlisting}
