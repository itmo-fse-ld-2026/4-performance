\subsection{Цель \texttt{diff}}

\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
<target name="diff">
  <echo message="Checking for changes in target classes..." />
  <java classname="org.tmatesoft.svn.cli.SVN"
        outputproperty="fse.diff.out"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="status"/>
  </java>

  <loadresource property="fse.diff.commitFiles">
    <string value="${fse.diff.out}"/>
    <filterchain>
      <linecontainsregexp>
        <regexp pattern="${fse.diff.classes}"/>
      </linecontainsregexp>
      <tokenfilter>
        <replaceregex pattern="^.[\s]+" replace="" flags="g"/>
      </tokenfilter>
      <striplinebreaks/>
      <trim/>
    </filterchain>
  </loadresource>

  <condition property="fse.diff.detected">
    <matches string="${fse.diff.out}" pattern="${fse.diff.classes}"/>
  </condition>

  <antcall target="-svn-commit-if-changed"/>
  <echo message="Done task 'diff'!" />
</target>

<target name="-svn-commit-if-changed" if="fse.diff.detected">
  <echo message="Target classes modified. Committing to SVN..."/>

  <tempfile property="fse.diff.targetsFile" 
            prefix=".commit-" 
            suffix=".txt"
            deleteonexit="true"/>
  
  <echo file="${fse.diff.targetsFile}" message="${fse.diff.commitFiles}"/>

  <java classname="org.tmatesoft.svn.cli.SVN"
        failonerror="true"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="commit"/>
    <arg value="-m"/>
    <arg value="feat: changes in tracked classes (Ant-automated)"/>
    <arg value="--targets"/>
    <arg value="${fse.diff.targetsFile}"/>
  </java>
</target>
\end{lstlisting}
