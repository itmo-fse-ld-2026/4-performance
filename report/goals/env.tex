\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
<target name="env">
  <echo message="Building project for specified Java environment..." />

  <antcall target="build">
    <param name="fse.compile.jvm.source" value="${fse.env.jvm.source}" />
    <param name="fse.compile.jvm.target" value="${fse.env.jvm.target}" />

    <param name="fse.build.version" value="${fse.build.version}-env" />
  </antcall>
  
  <echo message="JVM Arguments: ${fse.env.jvm.args}" />

  <exec executable="make">
    <env key="POSTGRES_HOST" value="${fse.env.postgres.host}"/>
    <env key="POSTGRES_DB" value="${fse.env.postgres.db}"/>
    <env key="POSTGRES_PORT" value="${fse.env.postgres.port}"/>
    <env key="POSTGRES_USER" value="${fse.env.postgres.user}"/>
    <env key="POSTGRES_PASSWORD" value="${fse.env.postgres.password}"/>
    <env key="VERSION_TAG" value="${fse.env.docker.tag}"/>
    <env key="JAVA_OPTS" value="${fse.env.jvm.args}"/>
    <env key="JAVA_HOME" value="${fse.env.jvm.home}"/>

    <arg value="run"/>
  </exec>
</target>
\end{lstlisting}
