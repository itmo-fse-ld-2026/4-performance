\subsection{Цель \texttt{build}}

\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
<target name="build" depends="compile">
  <echo message="Building the project..."/>

  <delete dir="${fse.build.target}" />
  <mkdir dir="${fse.build.target}" />
  
  <copy todir="${fse.build.target}">
    <fileset dir="${fse.build.webapp}" />
  </copy>
  <copy todir="${fse.build.target}/WEB-INF/classes">
    <fileset dir="${fse.compile.target}" />
  </copy>
  <copy todir="${fse.build.target}/WEB-INF/lib">
    <fileset dir="${fse.ant.home}/lib" />
  </copy>

  <jar destfile="${fse.build.target}/${fse.build.name}-${fse.build.version}.war"
        basedir="${fse.build.source}">
    <fileset dir="${fse.build.target}" />
    <manifest>
      <attribute name="Main-Class" value="${fse.build.main}"/>
      <attribute name="Specification-Title" value="${fse.build.title}"/>
      <attribute name="Specification-Version" value="${fse.build.version}"/>
      <attribute name="Specification-Vendor" value="${fse.build.vendor}"/>
    </manifest>
  </jar>

  <echo message="Done task 'build'!" />
</target>
\end{lstlisting}
