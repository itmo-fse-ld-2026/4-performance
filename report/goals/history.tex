\subsubsection{Листинг}

\begin{lstlisting}[caption={Часть содержимого файла \texttt{build.xml}},captionpos=b]
<target name="history">
  <delete file="${java.io.tmpdir}/${fse.history.broken_rev}" failonerror="true"/>

  <java classname="org.tmatesoft.svn.cli.SVN"
        fork="true"
        classpathref="fse.ivy.path.ant"
        dir="${basedir}">
    <arg value="cleanup"/>
  </java>

  <java classname="org.tmatesoft.svn.cli.SVN"
        outputproperty="fse.diff.out"
        failonerror="true"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="update"/>
  </java>

  <antcall target="-history-recursion-start" />

  <delete file="${java.io.tmpdir}/${fse.history.broken_rev}" failonerror="true"/>

  <echo message="Done task 'history'!"/>
</target>

<target name="-history-recursion-start">
  <trycatch>
    <try>
      <antcall target="compile"/>
      <antcall target="-history-create-log" />
    </try>
    <catch>
      <property name="fse.history.hasErrors" value="true"/>
      <antcall target="-history-check-if-last" />
    </catch>
  </trycatch>
</target>

<target name="-history-check-if-last">
  <echo message="[ERR] Failed to compile project! Checking previous revisions..." />

  <java classname="org.tmatesoft.svn.cli.SVN"
        outputproperty="fse.history.headRev"
        classpathref="fse.ivy.path.ant"
        failonerror="true"
        fork="true">
    <arg value="info"/>
    <arg value="--show-item"/>
    <arg value="revision"/>
  </java>

  <echo file="${java.io.tmpdir}/${fse.history.broken_rev}"
        message="${fse.history.headRev}" />

  <condition property="fse.history.isNotLast">
    <not>
      <equals arg1="${fse.history.headRev}" arg2="2"/>
    </not>
  </condition>

  <antcall target="-history-revert" />
</target>

<target name="-history-revert" if="fse.history.isNotLast" >
  <echo message="Found previous revision for r${fse.history.headRev}! Reverting..." />

  <java classname="org.tmatesoft.svn.cli.SVN"
        failonerror="true"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="update"/>
    <arg value="-r"/>
    <arg value="PREV"/>
  </java>

  <antcall target="-history-recursion-start" inheritAll="false" />
</target>

<target name="-history-create-log" if="${fse.history.hasErrors}" >
  <echo message="Found working version! Logging breaking changes..." />

  <loadfile property="fse.broken.rev" srcFile="${java.io.tmpdir}/${fse.history.broken_rev}"/>

  <java classname="org.tmatesoft.svn.cli.SVN" 
        outputproperty="fse.history.workingRev"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="info"/>
    <arg value="--show-item"/>
    <arg value="revision"/>
  </java>

  <java classname="org.tmatesoft.svn.cli.SVN" 
        output="${fse.history.log}"
        classpathref="fse.ivy.path.ant"
        fork="true">
    <arg value="diff"/>
    <arg value="-r"/>
    <arg value="${fse.history.workingRev}:${fse.broken.rev}"/>
  </java>
</target>
\end{lstlisting}
